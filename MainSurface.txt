package Vmain;
import Vcore.CoreInterface;
import Vcore.CoreProduct;
import java.util.Scanner;

public class MainSurface {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        CoreProduct product = new CoreProduct("Lakshimi",240);
        product.ShowProduct();
        System.out.print("Type the value for pay: ");
        double value = sc.nextDouble();
        CoreInterface.Pix pix = new CoreInterface.Pix();
        pix.pay(value, product);
    }
}
