package Umain;
import Ucore.CorePaymentFreeMarket;
import Ucore.CoreProducts;
import java.util.Scanner;


public class AmainExe {
    public static void main(String[] Args) {
        Scanner sc = new Scanner(System.in);
        CoreProducts p1 = new CoreProducts("Teclado Redragon Lakshimi", "Preto", 220);
        CoreProducts p2 = new CoreProducts("Mouse havit h2002s", "preto", 159.8);
        CoreProducts p3 = new CoreProducts("Headset Redragon acrux", "preto", 250.5);
        int ni;
        while (true) {
            System.out.println("--------- Products ---------");
            System.out.println("1 - " + p1.Products());
            System.out.println("2 - " + p2.Products());
            System.out.println("3 - " + p3.Products());
            System.out.println("Chose the product: ");
            ni = sc.nextInt();
            sc.nextLine();

            if (ni == 1) {
                System.out.println(p1.Products());
                break;
            } else if (ni == 2) {
                System.out.println(p2.Products());
                break;
            } else if (ni == 3) {
                System.out.println(p3.Products());
                break;
            } else if (ni == 5) {
                System.out.println("You exited.");
                break;
            } else {
                System.out.println("Unknown product.");
            }
        }
            CorePaymentFreeMarket.Payment payment;

            System.out.println("Chose one method of paying: ");
            String method=sc.nextLine().toLowerCase();
            switch (method){
                case "pix":
                    payment=new CorePaymentFreeMarket.Pix();
                    break;
                case "credit card": 
                    payment=new CorePaymentFreeMarket.CreditCard();
                    break;
                case "debit card":
                    payment=new CorePaymentFreeMarket.DebitCard();
                case "ticket":
                    payment =new CorePaymentFreeMarket.Ticket();
                    break;
                default:
                    System.out.println("Inválid method payment.");
                    return;
            }
            System.out.println("Type the value of pay: ");
            double value = sc.nextDouble();
            payment.pay(value);
            sc.close();
    }
}
