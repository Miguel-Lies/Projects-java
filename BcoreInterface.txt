package Vcore;

public class CoreInterface {
    public interface Payment {
        boolean pay(double value, CoreProduct product);
    }
             public static class Pix implements Payment {
        @Override
                 public boolean pay(double value, CoreProduct product) {
                     if (value >=product.getPrice()) {
                         System.out.println("Pay.");
                         return true;
                     } else {
                         System.out.println("Don't pay");
                         return false;
                     }
                 }
             }
             public static class DebitCard implements Payment{
        @Override
        public boolean pay(double value, CoreProduct product){

            if (value>=product.getPrice()){
                System.out.println("Pay.");
                return true;
            }else{
                System.out.println("Dont't pay.");
                return false;
            }
        }
    }
    public static class CreditCard implements Payment{
        public double available=1550;
        @Override
        public boolean pay(double value, CoreProduct product){
            if(value<product.getPrice()){
                System.out.println("Nonexistent balance.");
                return false;
            }
            else if (value<=available){
                available=available-value;
                return true;
            }else{
                System.out.println("Don't pay.");
                return false;
            }
        }
    }
    public static class Ticket implements Payment{
        @Override
        public boolean pay(double value, CoreProduct product){
            if (value>= product.getPrice()){
                System.out.println("Pay.");
                return true;
            }else{
                System.out.println("Don't pay.");
                return false;
            }
        }
    }
}
